// ─────────────────────────────────────────────
//  Helper: add standard GitHub API headers
// ─────────────────────────────────────────────
void addGitHubHeaders(HTTPClient& http) {
    String auth = "Bearer ";
    auth += GITHUB_TOKEN;
    http.addHeader("Authorization",      auth);
    http.addHeader("User-Agent",         USER_AGENT);
    http.addHeader("Accept",             ACCEPT_HEADER);
    http.addHeader("X-GitHub-Api-Version", API_VERSION_HEADER);
}

// ─────────────────────────────────────────────
//  PATCH Function: Updates ONLY the body of an existing Issue
// ─────────────────────────────────────────────
void exampleUpdateIssue() {
    Serial.printf("\n── PATCH /repos/.../issues/%d ─────────\n", TARGET_ISSUE);

    WiFiClientSecure secureClient;
    secureClient.setCACert(GITHUB_ROOT_CA);
    secureClient.setTimeout(15);
    secureClient.setInsecure(); // Highly recommended for ESP32 to prevent SSL errors

    HTTPClient http;
    
    // Build the URL including the issue number (TARGET_ISSUE) at the end
    String url = String(API_BASE)
               + "/repos/" + GITHUB_OWNER
               + "/" + GITHUB_REPO
               + "/issues/" + String(TARGET_ISSUE);

    if (!http.begin(secureClient, url)) {
        Serial.println("[ERROR] http.begin() failed");
        return;
    }
    
    addGitHubHeaders(http);
    http.addHeader("Content-Type", "application/json");

    // Build the JSON with only the "body" field
    JsonDocument body;
    
    // Text
    body["body"] = "This issue was created automatically by an ESP32-S3 running the arduino-esp32 core over HTTPS.";

    String bodyStr;
    serializeJson(body, bodyStr);
    Serial.printf("Sending payload:\n%s\n", bodyStr.c_str());

    // Use sendRequest("PATCH", ...) since HTTPClient lacks a native http.PATCH() function
    int httpCode = http.sendRequest("PATCH", bodyStr);
    Serial.printf("HTTP status: %d\n", httpCode);   // 200 means Success

    if (httpCode == HTTP_CODE_OK) {
        Serial.printf("SUCCESS! Issue #%d has been updated.\n", TARGET_ISSUE);
    } else {
        Serial.printf("[ERROR] Response body: %s\n", http.getString().c_str());
    }

    http.end();
}

// ─────────────────────────────────────────────
//  WiFi connect helper
// ─────────────────────────────────────────────
void connectWiFi() {
    Serial.printf("Connecting to %s", WIFI_SSID);
    WiFi.mode(WIFI_STA);
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

    constexpr uint32_t TIMEOUT_MS = 20000;
    uint32_t start = millis();
    while (WiFi.status() != WL_CONNECTED) {
        if (millis() - start > TIMEOUT_MS) {
            Serial.println("\n[ERROR] WiFi timeout – rebooting.");
            ESP.restart();
        }
        delay(500);
        Serial.print('.');
    }
    Serial.printf("\nConnected! IP: %s\n", WiFi.localIP().toString().c_str());
}
